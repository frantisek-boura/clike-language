int factorial(int number) {
    if (number == 0) {
        return 1;
    } else if (number == 1) {
        return 1;
    }  

    return number * factorial(number - 1);
}

print("Enter a number: ");
int num = s_to_i(read());
int result = factorial(num);
print("Factorial of " + i_to_s(num) + " is: " + i_to_s(result));
